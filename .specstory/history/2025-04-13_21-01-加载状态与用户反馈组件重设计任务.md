+ /**
+  * Animation utilities for loading and feedback components
+  */
+ 
+ // Helper to add animation class and remove it after animation completion
+ export const animateElement = (element, animationClass, duration = 1000) => {
+   if (!element) return;
+   
+   element.classList.add(animationClass);
+   
+   setTimeout(() => {
+     element.classList.remove(animationClass);
+   }, duration);
+ };
+ 
+ // Create a delayed sequence of animations
+ export const animateSequence = (elements, animationClass, delay = 200) => {
+   if (!elements || !elements.length) return;
+   
+   elements.forEach((element, index) => {
+     setTimeout(() => {
+       animateElement(element, animationClass);
+     }, index * delay);
+   });
+ };
+ 
+ // Progressive loading animation for skeleton elements
+ export const progressiveLoad = (containerSelector, itemSelector, delay = 100) => {
+   const container = document.querySelector(containerSelector);
+   if (!container) return;
+   
+   const items = container.querySelectorAll(itemSelector);
+   animateSequence(items, 'fade-in', delay);
+ };
+ 
+ // Subtle feedback animation for interactive elements
+ export const feedbackAnimation = (element) => {
+   animateElement(element, 'button-press', 300);
+ };
+ 
+ // Gold shimmer effect for luxury elements
+ export const addGoldShimmer = (element) => {
+   if (!element) return;
+   
+   element.classList.add('gold-shimmer');
+   
+   // Remove and re-add the class to restart animation when needed
+   element.addEventListener('mouseenter', () => {
+     element.classList.remove('gold-shimmer');
+     void element.offsetWidth; // Force reflow
+     element.classList.add('gold-shimmer');
+   });
+ };
+ 
+ // Create smooth transition between loading and content
+ export const transitionFromLoading = (loadingElement, contentElement, duration = 500) => {
+   if (!loadingElement || !contentElement) return;
+   
+   loadingElement.classList.add('fade-out');
+   
+   setTimeout(() => {
+     loadingElement.style.display = 'none';
+     contentElement.style.display = 'block';
+     contentElement.classList.add('fade-in');
+   }, duration);
+ };
+ 
+ // Toast notification show/hide with animations
+ export const showToast = (toastElement, duration = 3000) => {
+   if (!toastElement) return;
+   
+   toastElement.classList.add('show');
+   toastElement.classList.add('toast-slide-in');
+   
+   setTimeout(() => {
+     toastElement.classList.remove('toast-slide-in');
+     toastElement.classList.add('toast-slide-out');
+     
+     setTimeout(() => {
+       toastElement.classList.remove('show');
+       toastElement.classList.remove('toast-slide-out');
+     }, 500);
+   }, duration);
+ };
+ 
+ // Animate progress bar
+ export const animateProgress = (progressElement, targetValue, duration = 1000) => {
+   if (!progressElement) return;
+   
+   const startValue = parseInt(progressElement.style.width) || 0;
+   const startTime = performance.now();
+   
+   const updateProgress = (currentTime) => {
+     const elapsedTime = currentTime - startTime;
+     const progress = Math.min(elapsedTime / duration, 1);
+     const currentValue = startValue + (targetValue - startValue) * progress;
+     
+     progressElement.style.width = `${currentValue}%`;
+     
+     if (progress < 1) {
+       requestAnimationFrame(updateProgress);
+     }
+   };
+   
+   requestAnimationFrame(updateProgress);
+ };

+     // Hide with animation
+     this.container.classList.remove('visible');
+     
+     this.isOpen = false;
+     
+     // Wait for animation to complete before removing from DOM
+     setTimeout(() => {
+       // Unlock body scroll
+       document.body.classList.remove('luxury-dialog-open');
+       
+       this.container.style.visibility = 'hidden';
+       
+       // Call onClose callback if provided
+       if (this.closeCallback) {
+         this.closeCallback();
+       }
+     }, 300);
+     
+     return this;

+         contentElement.innerHTML = '';
+         
+         if (typeof content === 'string') {
+           contentElement.innerHTML = content;
+         } else if (content instanceof Node) {
